[문제 1] 배열 원소의 값을 오름차순 또는 내림차순으로 정렬하기 위한 프로그램을 작성하세요.

아래에 선언된 함수를 정의합니다.

#include <stdio.h>

// 정수 포인터 두 개를 전달 받아 두 값을 교환하는 함수
void intSwap(int *x, int *y);

// 정수 배열과 배열 원소의 수를 전달받아 배열 원소의 값을 출력하는 함수
void intPrinter(int *arr, int count);

// 두 정수 값을 비교하고 값의 차이를 반환하는 함수
int intCompareAsc(int *x, int *y);

// 두 정수 값을 비교하고 값의 차이를 반환하는 함수
int intCompareDesc(int *x, int *y);

// 정수 배열과 배열 원소의 수, 비교함수를 전달 받아 첫 번째 인수로 전달받은 배열 원소를 정렬합니다.
// 정렬에 사용할 알고리즘은 제한하지 않습니다. 삽입, 버블, 선택정렬 중 하나를 사용합니다.
void intSort(int* arr, int count, int (*compareFunc)(int*, int*));

int main() {
	int intAry[] = { 6, 3, 5, 7, 8, 2, 1, 9, 10, 4 };
	int count = (int)(sizeof(intAry) / sizeof(intAry[0]));
	print(intAry, count);
	intSort(intAry, count, intCompareAsc);
	print(intAry, count);
	intSort(intAry, count, intCompareDesc);
	print(intAry, count);
	return 0;
}
